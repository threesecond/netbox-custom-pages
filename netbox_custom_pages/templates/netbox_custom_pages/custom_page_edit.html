fetch('/api/plugins/custom-pages/proxy/', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-CSRFToken': document.cookie.split('csrftoken=')[1].split(';')[0]
  },
  body: JSON.stringify({
    target_url: 'https://grafana.internal/api/dashboards',
    token_key: 'grafana_read_only', // Alias mapped in configuration.py
    method: 'GET'
  })
})
  .then(res => res.json())
  .then(data => console.log(data.data)); // Result payload is in data.data
